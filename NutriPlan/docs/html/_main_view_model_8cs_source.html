155 private string ResolveProductsJsonPath()
156 {
157 // Try application base directory relative paths and project-relative Data folder
158 var candidates = new[]
159 {
160 Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Data", "products.json"),
161 Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "..", "..", "..", "Data", "products.json"),
162 Path.Combine(Environment.CurrentDirectory, "Data", "products.json"),
163 Path.Combine(Directory.GetCurrentDirectory(), "Data", "products.json"),
164 Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "products.json"),
165 "Data\\products.json",
166 "products.json"
167 };
168
169 foreach (var c in candidates)
170 {
171 var full = Path.GetFullPath(c);
172 if (File.Exists(full)) return full;
173 }
174
175 // Fallback: return relative path as given (may not exist)
176 return "Data\\products.json";
177 }

182 private void Generate()
183 {
184 // Validate inputs
185 if (Weight <= 0 || Height <= 0 || Age <= 0) return;
186 if (NumberOfMeals < 2 || NumberOfMeals > 5) return;
187
188 // 1. Calculate BMR
189 double bmr = _calorieCalculator.CalculateBmr(Weight, Height, Age, SelectedGender);
190
191 // 2. Calculate TDEE
192 double tdee = _calorieCalculator.CalculateTdee(bmr, SelectedActivity);
193
194 // 3. Calculate target calories based on goal
195 double targetCalories = _calorieCalculator.CalculateTargetCalories(tdee, SelectedGoal);
196
197 // 4. Calculate daily macro distribution (B/F/C)
198 MacroResult = _macroCalculator.CalculateMacros(Weight, targetCalories, SelectedGoal);
199
200 // 5-6. Generate meals and products
201 Dictionary<string, List<Product>> mealProducts;
202 Meals = _mealDistributor.GenerateMealPlan(_productRepository, MacroResult, NumberOfMeals, out mealProducts);
203 MealProducts = mealProducts;
204 }
