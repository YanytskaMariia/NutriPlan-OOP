38 [TestMethod]
39 public void SequentialAndParallel_TotalsWithinTolerance()
40 {
41 var repo = new JsonProductRepository(ProductsPath) as IProductRepository;
42 var macros = new MacroResult { TotalCalories = 2000, ProteinGrams = 150, FatGrams = 70, CarbsGrams = 250 };
43 int mealCount = 3;
44
45 var mealService = new MealDistributionService();
46 Dictionary<string, List<Product>> seqMap;
47 var seqMeals = mealService.GenerateMealPlan(repo, macros, mealCount, out seqMap);
48
49 var parallelService = new ParallelMealGenerationService();
50 Dictionary<string, List<Product>> parMap;
51 var parMeals = parallelService.GenerateMealPlanParallel(repo, macros, mealCount, out parMap, degreeOfParallelism: Environment.ProcessorCount);
52
53 var seqTotal = seqMap.Values.SelectMany(x => x).Sum(p => p.Calories);
54 var parTotal = parMap.Values.SelectMany(x => x).Sum(p => p.Calories);
55
56 var diff = Math.Abs(seqTotal - parTotal);
57 var rel = seqTotal > 0 ? diff / seqTotal : diff;
58
59 // Allow 10% tolerance due to greedy/randomized selection differences
60 Assert.IsTrue(rel <= 0.10, $"Totals differ more than 10%: seq={seqTotal}, par={parTotal}");
61 }
